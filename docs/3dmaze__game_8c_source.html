317 void MoveCamera(float dx, float dy) {

324  game.camera.location.x += dy * -s;

335 void Receiver(uint8_t c) {
336  switch (c) {
337  case 'w':
338  case 'W':
339  MoveCamera(CAMERA_MOVE, 0);
340  RenderWorld();
341  CheckWin();
342  break;
343  case 's':
344  case 'S':
345  MoveCamera(-CAMERA_MOVE, 0);
346  RenderWorld();
347  CheckWin();
348  break;
349  case 'a':
350  case 'A':
351  MoveCamera(0, CAMERA_MOVE);
352  RenderWorld();
353  CheckWin();
354  break;
355  case 'd':
356  case 'D':
357  MoveCamera(0, -CAMERA_MOVE);
358  RenderWorld();
359  CheckWin();
360  break;
361  case '<':
362  case ',':
363  game.camera.rotation.z += CAMERA_ROTATE;
364  RenderWorld();
365  break;
366  case '>':
367  case '.':
368  game.camera.rotation.z -= CAMERA_ROTATE;
369  RenderWorld();
370  break;
371  case '\r':
372  //GameOver();
373  break;
374  default:
375  break;
376  }
